Quick Start

üì• Pull the Image

# Pull latest (PHP 8.4)
docker pull almossaidllc/php-laravel-base:latest

# Or specific PHP version
docker pull almossaidllc/php-laravel-base:8.3
docker pull almossaidllc/php-laravel-base:8.2

üöÄ Use in Your Dockerfile

FROM almossaidllc/php-laravel-base:8.3

WORKDIR /var/www/html

# Copy application
COPY . .

# Install dependencies
RUN composer install --no-dev --optimize-autoloader

# Set permissions
RUN chown -R appuser:appgroup storage bootstrap/cache

USER appuser

CMD ["php-fpm"]

üé® Create New Laravel Project

docker run --rm -v $(pwd):/app -w /app \
    almossaidllc/php-laravel-base:8.3 \
    laravel new my-project
